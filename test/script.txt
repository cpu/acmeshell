echo "Creating an order for example.acmeshell.com"
newOrder -identifiers=example.acmeshell.com
echo "Solving the HTTP-01 challenge for the sole order authorization"
solve -challengeType=http-01 {{ (authz (order 0) \"example.acmeshell.com\") }}
echo "Polling the order until it is ready for finalization"
poll -order 0 -status=ready
echo "Finalizing the order with a CSR signed by a throw-away key"
finalize -order 0
echo "Polling the order until it is valid and has a certificate"
poll -order 0 -status=valid
echo "Fetching the order's certificate"
getCert -order 0
echo "All done!"
